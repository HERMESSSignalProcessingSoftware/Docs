\newcommand{\bitrect}[2]{
	\begin{pgfonlayer}{foreground}
		\draw [thick] (0,0) rectangle (#1,1);
		\pgfmathsetmacro\result{#1-1}
		\foreach \x in {1,...,\result}
		\draw [thick] (\x,1) -- (\x, 0.8);
	\end{pgfonlayer}
	%  \node [below left, align=right] at (0,0) {Type \\ Reset};
	\bitlabels{#1}{#2}
}

	}
